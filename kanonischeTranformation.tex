\begin{frame}
    \frametitle{Gesucht: Lösung der kanonischen Gleichungen}
    
    Gesucht werden die Lösungen $p_k$, $q_k$ zu
    \begin{align*}
        \dot{q}_k &= \frac{\partial H}{\partial p_k} \\
        \dot{p}_k &= -\frac{\partial H}{\partial q_k}.
    \end{align*}
    
    Jedoch: In dieser Form nicht lösbar.
    
\end{frame}
