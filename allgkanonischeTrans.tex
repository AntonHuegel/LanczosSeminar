\begin{frame}
    \frametitle{Die allgemeine kanonische Transformation}
    Die Invarianz von    
    \begin{displaymath}
            \sum_{i=1}^n p_i \partial q_i = \sum_{i=1}^n P_i \partial Q_i
    \end{displaymath}
    ist nicht notwendig. \\
    
    Allgemeinerer Ansatz:
    \begin{displaymath}
    \sum_{i=1}^n p_i \partial q_i = \sum_{i=1}^n P_i \partial Q_i + \partial S
    \end{displaymath}

\end{frame}

\end{frame}
