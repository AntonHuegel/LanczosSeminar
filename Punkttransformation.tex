\begin{frame}
    \frametitle{Punkttransformation}
    
    Alte Koordinaten: $p_k, q_k$ \\
    Neue Koordinaten: $P_k, Q_k$
    
    Punkttransformationen haben die Form
    
    \begin{align*}
    q_1 &= f_1(Q_1,\ldots,Q_n) \\
        &\quad\vdots \\
    q_n &= f_n(Q_1,\ldots,Q_n)    
    \end{align*}
    
\end{frame}
